87  // Get the scene to transition to
88  string getNewScene() {
89  lua::State *L = uiEngine.getState();
90  try {
91  LuaRef sceneTable = getGlobal(L->getState(), uiName.c_str());
92  if (!sceneTable.isTable()) // Check to see if the scene table exists
93  throw new std::exception("Scene table global not found");
94 
95  // Find the callback function we wantin the FunctionCallbacks table
96  LuaRef transTable = sceneTable["SceneTransition"];
97  if (!transTable.isTable())
98  throw new std::exception("SceneTransition table not found");
99 
100  LuaRef v = transTable["To"];
101  if (!v.isString())
102  throw new std::exception("Couldn't find 'to' value of table");
103  return v.tostring();
104  }
105  catch (std::exception& e) {
106  string what = e.what();
107  dlua.error("Failed Scene Transition info get (" + uiName + "): " + what);
108  }
109  return "";
110  }
